        function togglePeriod(index) {
            const content = document.getElementById(`content-${index}`);
            const icon = document.getElementById(`icon-${index}`);
            
            if (content.classList.contains('hidden')) {

                icon.classList.add('rotate-180');
            } else {
                content.classList.add('hidden');
                icon.classList.remove('rotate-180');
            }

        function calcularPresencial() {
            const v1 = parseFloat(document.getElementById('nota-v1').value) || 0;
            const vt = parseFloat(document.getElementById('nota-vt').value) || 0;
            const v2 = parseFloat(document.getElementById('nota-v2').value) || 0;
            const ms = ((v1 * 2) + vt + (v2 * 2)) / 5;
            
            const display = document.getElementById('media-display');
            const status = document.getElementById('status-display');
            const resultBox = document.getElementById('result-box');
            const vsBox = document.getElementById('vs-simulation');
            
            resultBox.classList.remove('hidden');
            display.innerText = ms.toFixed(1);

            if (ms >= 7.0) {
                status.innerText = "APROVADO";
                status.className = "inline-block px-6 py-2 rounded-full text-sm font-bold bg-[#46DA61] text-black";
                vsBox.classList.add('hidden');
            } else if (ms < 4.0) {
                status.innerText = "REPROVADO";
                status.className = "inline-block px-6 py-2 rounded-full text-sm font-bold bg-red-500 text-white";
                vsBox.classList.add('hidden');
            } else {
                status.innerText = "VERIFICAÇÃO SUPLEMENTAR";
                status.className = "inline-block px-6 py-2 rounded-full text-sm font-bold bg-orange-500 text-white";
                
                // Calculo VS (Standard Media 5.0 final)
                // (Media + VS) / 2 = 5.0 -> VS = 10 - Media
                let needed = (10 - ms);
                vsBox.classList.remove('hidden');
                document.getElementById('vs-needed').innerText = needed.toFixed(1);
            }
        }

        function calcularCR() {
            const notas = document.querySelectorAll('.cr-nota');
            const chs = document.querySelectorAll('.cr-ch');
            let totalWeightedScore = 0;
            let totalHours = 0;

            notas.forEach((input, index) => {
                const n = parseFloat(input.value) || 0;
                const h = parseFloat(chs[index].value) || 0;
                if(h > 0) {
                    totalWeightedScore += (n * h);
                    totalHours += h;
                }
            });

            const resultBox = document.getElementById('cr-result-box');
            const display = document.getElementById('cr-display');
            resultBox.classList.remove('hidden');

            if (totalHours > 0) {
                const cr = totalWeightedScore / totalHours;
                display.innerText = cr.toFixed(2);
            } else {
                display.innerText = "0.00";
            }
        }
